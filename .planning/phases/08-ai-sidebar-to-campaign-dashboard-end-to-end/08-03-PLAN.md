    ```typescript
    const renderAssetPreview = (item: DrillDownItem<Asset>): PreviewDescriptor => {
      const latestIter = latestIterationByAssetId[item.id];
      if (latestIter?.htmlPath && latestIter.generationStatus === 'complete') {
        const dims = getAssetDimensions(item.data.assetType);
        return { src: `/api/iterations/${latestIter.id}/html`, ...dims };
      }
      // Fallback for pending/generating
      return {
        width: 1080, height: 1080,
        meta: {
          icon: 'asset',
          badges: [item.data.assetType, latestIter?.generationStatus ?? 'pending'],
          lines: [item.data.title]
        }
      };
    };
    ```

    Add a `getAssetDimensions` helper function:
    ```typescript
    function getAssetDimensions(assetType: string): { width: number; height: number } {
      const map: Record<string, { width: number; height: number }> = {
        instagram: { width: 1080, height: 1080 },
        linkedin: { width: 1200, height: 627 },
        'one-pager': { width: 816, height: 1056 },
      };
      return map[assetType] ?? { width: 1080, height: 1080 };
    }
    ```

    ```typescript
    const renderFramePreview = (item: DrillDownItem<Frame>): PreviewDescriptor => {
      // iterations array from store is already loaded when at frame level
      const frameIterations = iterations.filter(i => i.frameId === item.id);
      const latest = frameIterations[frameIterations.length - 1];
      if (latest?.htmlPath && latest.generationStatus === 'complete') {
        // Get parent asset's type for dimensions
        const parentAsset = assets.find(a => a.id === item.data.assetId);
        const dims = getAssetDimensions(parentAsset?.assetType ?? 'instagram');
        return { src: `/api/iterations/${latest.id}/html`, ...dims };
      }
      return {
        width: 1080, height: 1080,
        meta: { icon: 'frame', badges: [`Slide ${item.data.frameIndex + 1}`], lines: [] }
      };
    };
    ```
